create table "schemaOsfix".products
(
    "productId"      bigserial
        constraint products_pk
            primary key
        constraint products_application_applicationid_fk
            references "schemaOsfix".application
            on update cascade on delete cascade,
    "productName"    text    not null,
    "productWeight"  numeric not null,
    "productComment" text
);

create unique index products_productid_uindex
    on "schemaOsfix".products ("productId");
